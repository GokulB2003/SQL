78.Full Report (Student + Subject + Questions Count + Marks)
SELECT
    s.student_name,
    sb.subject_name,
    COUNT(DISTINCT sq.question_id) AS total_questions,
    SUM(r.marks_obtained) AS marks_obtained
FROM student s
JOIN result r
    ON s.student_id = r.student_id
JOIN ExamSchedule ex
    ON r.exam_id = ex.exam_id
JOIN subject sb
    ON ex.subject_id = sb.subject_id
LEFT JOIN SubjectQuestion sq
    ON sb.subject_id = sq.subject_id
GROUP BY
    s.student_name,
    sb.subject_name;
